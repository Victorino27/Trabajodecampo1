using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;

namespace problema_n._03_TC_2_FCC
{
    internal class Program
    {
        static void Main(string[] args)
        {
            int num;
            string linea;
            Console.Write("Ingrese un numero: ");
            linea = Console.ReadLine();
            num = int.Parse(linea);
            if (num % 2 == 0)
            {
                Console.WriteLine("\n{0} es un numero par", num);
            }else{
                Console.WriteLine("\n{0} es un numero impar", num);
            }
            Console.ReadKey();
        }
    }
}
