    internal class Program

        {
            int sala;
            string rng;
            string linea;

            Console.WriteLine("Escriba su salario mensual: ");
            linea = Console.ReadLine();
            sala = int.Parse(linea);

            if (sala < 1200)
            {
                        rng = "bajo";
            }
            else
            if (sala >= 1200 && sala <= 2500)
                {
                    rng = "medio";
                } 
             else
                {
                    rng = "alto";
                }

            Console.WriteLine("Su salario del mes es: " + rng);
            Console.ReadKey();
        }
    }
